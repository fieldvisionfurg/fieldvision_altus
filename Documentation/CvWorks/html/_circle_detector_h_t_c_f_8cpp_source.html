61 void CircleDetectorHTCF::colorFilter(const cv::Mat& src, cv::Mat& dst) const
62 {
63  using namespace cv;
64  // http://docs.opencv.org/trunk/doc/py_tutorials/py_imgproc/py_colorspaces/py_colorspaces.html
65 
66  // Converte para HSV
67  Mat hsv;
68  cvtColor(src,hsv,CV_BGR2HSV);
69 
70  // Filtra
71  Scalar minValue(std::max(0,meanHue-varHue),50,50);
72  Scalar maxValue(std::min(179,meanHue+varHue),255,255);
73  inRange(hsv,minValue,maxValue,dst);
74 }

76 std::vector<Circle<>> CircleDetectorHTCF::detect(const cv::Mat& img) const
77 {
78  using namespace cv;
79 
80  // IMAGE PRE-PROCESSING
81  Mat gray,blured,dst;
82  const double sigma = img.rows * gaussFilterSigma;
83  if(useBlur && useColorFilter){
84  //apply blur first, then color filter
85  GaussianBlur( img, blured, Size(0,0), sigma, sigma );
86  colorFilter(blured,dst);
87  }
88  if(!useBlur && useColorFilter){
89  colorFilter(img,dst);
90  }
91  if(useBlur && !useColorFilter){
92  if(img.channels()>1)
93  cvtColor(img, gray, CV_BGR2GRAY);
94  else
95  gray=img;
96  GaussianBlur( gray, dst, Size(0,0), sigma, sigma );
97  }
98  if(!useBlur && !useColorFilter){
99  if(img.channels()>1)
100  cvtColor(img, dst, CV_BGR2GRAY);
101  else
102  dst=img;
103  }
104 
105  if(showProcessingImgs){ imshow("Pre-processed image",dst); cvWaitKey(1);}
106 
107  // APPLY HOUGH TRANSFORM
108  vector<Vec3f> circles;
109  HoughCircles(dst, circles, CV_HOUGH_GRADIENT,
110  2, dst.rows/8, houghParam1, houghParam2 );
111 
112  std::vector<Circle<>> lc;
113  for(const Vec3f& c : circles)
114  lc.push_back(Circle<>((int)c[0],(int)c[1],(int)c[2]));
115  return lc;
116 }
117 }
