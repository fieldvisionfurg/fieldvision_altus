47 void FireTrackerSA::update(const VisionCore::Frame<cv::Mat>& frame) {
48  m_currentTrack.clear();
49  spatial_data.clear();
50  current_contours.clear();
51  cv::Mat img = frame.getImg();
52 
54  if(img.size() == last_frame.size()){
55  cv::absdiff(img, last_frame, motion);
56  cv::cvtColor(motion, motion, CV_BGR2GRAY);
57  cv::threshold(motion, motion, 5, 255, CV_THRESH_BINARY);
58  cv::erode(motion, motion, cv::Mat::ones(cv::Size(2,2), CV_8UC1));
59  }else{
60  motion = cv::Mat::zeros(img.size(), CV_8UC1);
61  }
62  last_frame = img.clone();
63 
64 
66  cv::Mat img_hsv;
67  cv::cvtColor(img, img_hsv, CV_BGR2HSV);
68 
69  std::vector<std::vector<float>> channels_hsv, channels_rgb;
71  img.convertTo(img, CV_32FC3);
72  img_hsv.convertTo(img_hsv, CV_32FC3);
73  TemporalDataExtractor tde;
75  channels_hsv = tde.mat2vector(img_hsv);
76  channels_rgb = tde.mat2vector(img);
78  cv::Mat color_data = cv::Mat::zeros(img_hsv.rows*img_hsv.cols, 6, CV_32F); //6 is bgr + hsv
79 
80  for(int i = 0; i< color_data.rows; i++){
81  float *p = color_data.ptr<float>(i);
82  p[0] = channels_rgb[0][i]; //blue
83  p[1] = channels_rgb[1][i]; //green
84  p[2] = channels_rgb[2][i]; //red
85  p[3] = channels_hsv[0][i]; //hue
86  p[4] = channels_hsv[1][i]; //saturation
87  p[5] = channels_hsv[2][i]; //value
88  }
89 
91  std::vector<float> color_predictions;
92  color_predictions.resize(color_data.rows);
93 
94 
95  for(int i = 0; i< color_data.rows; i++){
97  cv::Mat r = color_data.row(i);
98  color_predictions[i] = (float) rfh_color.classify(r);
99  }
100 
101  color_classified = tde.vector2Mat(color_predictions, img.rows, img.cols);
102  color_classified.convertTo(color_classified, CV_8UC1);
103  //END of color classification
104 
106  std::vector<std::vector<cv::Point> > contours;
107  std::vector<cv::Vec4i> hierarchy;
108  cv::Mat canny_output = color_classified.clone();
109  cv::findContours( canny_output, contours, hierarchy, CV_RETR_EXTERNAL, CV_CHAIN_APPROX_NONE);
110 
112  getStatistics(img, contours);
113  //getTextureFeatures(img, contours);
114 
115  if(!dont_classify_by_region_stats){
116  if(spatial_data.size()>0){
117  cv::Mat region_data = cv::Mat::zeros(spatial_data.size(), spatial_data[0].size(), CV_32F);
118  for(int i = 0; i< region_data.rows; i++){
119  float *p = region_data.ptr<float>(i);
120  for(unsigned j = 0; j < spatial_data[0].size(); j++)
121  p[j] = spatial_data[i][j];
122  }
123 
124  std::vector<std::vector<cv::Point>> tmp_c;
125  std::vector<std::vector<float>> tmp_stat;
126  for(int i = 0; i< region_data.rows; i++){
127  //each region is classified using the classifier
128  cv::Mat r = region_data.row(i);
129  if(rfh_region.classify(r) > 0.0){
130  tmp_c.push_back(current_contours[i]);
131  tmp_stat.push_back(spatial_data[i]);
132  }
133  }
134  current_contours = tmp_c; spatial_data = tmp_stat;
135  }
136 
137 
139  cv::Mat current_binary = cv::Mat::zeros(frame.getImg().size(), CV_8UC1);
140  for(unsigned i = 0; i < current_contours.size() ; i++)
141  cv::drawContours(current_binary, current_contours, i, cv::Scalar::all(1), -1);
142 
143  if(current_binary.size() == last_binary.size()){
144  cv::Mat tmp;
145  std::vector<std::vector<cv::Point>> cmb_contours;
146  cv::bitwise_and(current_binary, last_binary, tmp);
147  cv::medianBlur(tmp, tmp, 5);
148  cv::findContours( tmp, cmb_contours, hierarchy, CV_RETR_EXTERNAL, CV_CHAIN_APPROX_NONE);
150  for(auto c :cmb_contours){
151  int area = (int) cv::contourArea(c);
152  if(area > 100){
153  cv::Scalar m = cv::sum( motion(boundingRect(c) ));
154  if(m[0] > 0.2 * area)
155  m_currentTrack.push_back(c);
156  }
157  }
159  m_currentTrack = cs.smoothContoursP(m_currentTrack, .01f);
160  }
161  last_binary = current_binary;
162  }
163  else{
164  m_currentTrack = current_contours;
165  }
167  if(show_debug_info){
168  cv::imshow("motion", motion);
169  cv::normalize(color_classified,color_classified,0,255, cv::NORM_MINMAX);
170  cv::imshow("color segmentation", color_classified);
171  }
172 }

287 void FireTrackerSA::getStatistics(cv::Mat &image, std::vector<std::vector<cv::Point>> &in){
288 
289  for( unsigned i = 0; i< in.size(); i++ ){ // for each contour
290  cv::Rect roi = cv::boundingRect(in[i]);
291  if(cv::contourArea(in[i])> 150 && roi.height > 3 && roi.width > 3){ //very small areas are ignored
301  std::vector<float> contour_data; contour_data.resize(24);
302 
303  float c_roundness;
304  float c_area = static_cast<float>(cv::contourArea(in[i]));
305  cv::Scalar c_mean, c_std_dev, c_median, c_skewness;
306 
307  c_roundness = getRoundness(in[i]);
308  //get data
309  //create mask
310  cv::Mat mask = cv::Mat::zeros(cv::Size(image.cols, image.rows), CV_8UC1);
311  drawContours( mask, in, i, cv::Scalar::all(1), -1);
312  //crop areas
313  mask = mask(roi);
314  cv::Mat the_region = image(roi).clone();
315 
316  //extract dispersion data
317  cv::meanStdDev(the_region, c_mean, c_std_dev, mask);
318 
319  // the median is used in order to compute the skewness
320  // the way it is implemented here is using histograms in order to find the center
321  // Assimetria de Pearson
322  // mean less median/ std deviation
323  c_median = getMedian(the_region, mask);
324 
325  c_skewness[0] = c_std_dev[0] > 0? 3*(c_mean[0] - c_median[0]) / c_std_dev[0] : 0;
326  c_skewness[1] = c_std_dev[1] > 0? 3*(c_mean[1] - c_median[1]) / c_std_dev[1] : 0;
327  c_skewness[2] = c_std_dev[2] > 0? 3*(c_mean[2] - c_median[2]) / c_std_dev[2] : 0;
328 
329  // get the mean for now subregions
330  cv::Scalar c_sr_mean, c_sr_std_dev;
331  for(unsigned c = 0; c < 3; c++){
332  for(unsigned r = 0; r < 3;r++){
333 
334  cv::Rect sr_roi(cvFloor(the_region.cols/3)*c, cvFloor(the_region.rows/3)*r, cvFloor(the_region.cols/3), cvFloor(the_region.rows/3));
335  cv::Mat s_region = the_region(sr_roi).clone(), s_mask = mask(sr_roi).clone();
336  cv::meanStdDev(s_region, c_sr_mean, c_sr_std_dev, s_mask);
337  // red channel only
338  contour_data[14+(3*c)+r] = static_cast<float>(c_sr_mean[2]/c_mean[2]);
339  }
340  }
341  //0 - area; 1 - roundness; 2,3,4 - mean; 5,6,7 - median; 8,9,10 - standard deviation; 11,12,13 - skewness
342  contour_data[0] = c_area;
343  contour_data[1] = c_roundness;
344  contour_data[2] = static_cast<float>(c_mean[0]); contour_data[3] = static_cast<float>(c_mean[1]); contour_data[4] = static_cast<float>(c_mean[2]);
345  contour_data[5] = static_cast<float>(c_median[0]); contour_data[6] = static_cast<float>(c_median[1]); contour_data[7] = static_cast<float>(c_median[2]);
346  contour_data[8] = static_cast<float>(c_std_dev[0]); contour_data[9] = static_cast<float>(c_std_dev[1]); contour_data[10] = static_cast<float>(c_std_dev[2]);
347  contour_data[11] = static_cast<float>(c_skewness[0]); contour_data[12] = static_cast<float>(c_skewness[1]); contour_data[13] = static_cast<float>(c_skewness[2]);
348  contour_data[23] = (float)the_region.cols/(float)the_region.rows;
349  //
350  spatial_data.push_back(contour_data);
351  current_contours.push_back(in[i]);
352  if (show_debug_info)
353  std::cout << "\nsigma = " << c_std_dev <<"\t skewness = " << c_skewness << "\t mean = " <<c_mean << "\tmedian = " <<c_median << "\tarea = " << c_area << "\troundness = " << c_roundness;
354 
355  }//endif
356  }//end contours for

362 cv::Scalar FireTrackerSA::getMedian(cv::Mat &image, cv::Mat &mask){
363 
364  cv::Scalar out;
365  std::vector<cv::Mat> img_channels;
366  cv::split(image, img_channels);
367 
368  for(unsigned i=0; i < img_channels.size(); i++){ //for each channel
369  cv::Mat cur_img = img_channels[i].clone(); cur_img.convertTo(cur_img, mask.type());
370  int histSize[] = {256}; float sranges[] = { 0, 256 }; const float* ranges[] = { sranges };
371  cv::Mat hist;
372  // we compute the histogram from the 0-th and 1-st channels
373  int channels[] = {0};
374  calcHist( &cur_img, 1, channels, mask,
375  hist, 1, histSize, ranges,
376  true, // the histogram is uniform
377  false );
378 
379  long sum = 0; int bin = 0;
380  cv::bitwise_and(cur_img, mask, cur_img);
381  cv::Scalar maxElements = cv::sum(cur_img);
382  // the bean where the midlle of the values is reached is the center point (or median)
383  while((sum < maxElements[0]/2) && (bin < 255))
384  {
385  sum += (long) hist.at<float>(bin);
386  bin ++;
387  }
388  out[i] = bin;
389  }
390  //return the median
391  return out;
392 }
