208 template<class TImg>
209 void AsyncFrameServerWrap<TImg>::threadExecution()
210 {
211  using namespace std::chrono;
212 
213  m_thrExeLock.lock();
214  high_resolution_clock::time_point captureTime;
215  milliseconds period;
216  while(m_frameServer->hasNext()){
217  if(m_finish)
218  break;
219  if( m_started){
220  // Store time of capture
221  captureTime = high_resolution_clock::now();
222 
223  // Capture frame and copy it to m_currentFrame;
224  lockCurrentFrame();
225  m_currentFrame = std::shared_ptr<Frame<TImg>>(new Frame<TImg>(m_frameServer->captureFrame()));
226  unlockCurrentFrame();
227 
228  // Callback execution passing captured frame
229  for(std::function<void(std::shared_ptr<Frame<TImg>>)>& fcn : m_callback)
230  fcn(m_currentFrame);
231  // Signal new frame to other threads
232  m_newFrameCondVar.notify_all();
233 
234  // Enforce specified frames per second (if setted > 0)
235  period=milliseconds(1000/(int)m_framesPerSecond); // period between frames
236  if(m_framesPerSecond>0){
237  milliseconds waitTime = duration_cast<milliseconds>(period-(high_resolution_clock::now()-captureTime));
238  if(waitTime.count()>0)
239  std::this_thread::sleep_for(waitTime);
240  }
241 
242  // If the frames to capture reached 0 then pause
243  if(m_framesToCapture>0)
244  m_framesToCapture--;
245  if(m_framesToCapture==0){
246  m_framesToCapture=-1;
247  m_started=false;
248  }
249  }
250  else{
251  // Wait untill start()
252  m_threadExecCondVar.wait(m_thrExeLock);
253  }
254  }
255  m_finished=true;
256  m_thrExeLock.unlock();
257 }

374 template<class TImg,class TOut>
375 void AsyncVisionExecution<TImg,TOut>::threadExecution()
376 {
377  m_thrExeLock.lock();
378  std::condition_variable& cv = m_asyncFrameServerPtr->getNewFrameCondVar();
379  while(m_asyncFrameServerPtr->finished()==false && m_finish==false){
380  //wait for new frame notification
381  cv.wait(m_thrExeLock);
382  while(!m_started && !m_finish)
383  cv.wait_for(m_thrExeLock,std::chrono::seconds(1));
384  if(m_finish)
385  break;
386 
387  //capture and process frame
388  m_asyncFrameServerPtr->lockCurrentFrame();
389  std::shared_ptr<Frame<TImg>> f = m_asyncFrameServerPtr->getCurrentFrame();
390  //TODO: improve performance: use read-write lock (not yet available in C++11)
391  m_asyncFrameServerPtr->unlockCurrentFrame();
392 
393  try{
394  TOut result = process(f);
395  //call callback, passing the result
396  if(m_callback!=NULL && m_callback)
397  m_callback(result);
398  }
399  catch(std::exception& e){
400  std::cout<<"Deu pau no processo. \n";
401  std::cout<<e.what();
402  std::cout.flush();
403  m_finish=true;
404  }
405  }
406  m_started=false;
407  m_thrExeLock.unlock();
408 }
