86 void DatasetFDDB::extractData()
87 {
88  m_samplesID.clear();
89  m_objects.clear();
90  for(const std::string& foldFileName : m_folds){
91  // Opens fold file
92  std::ifstream ifs (foldFileName, std::ifstream::in);
93 
94  // While has not reached end of file
95  while(ifs.good()){
96  // Read sample ID
97  std::string sampID;
98  ifs>>sampID;
99  if(sampID.length() > 1){
100  m_samplesID.push_back(sampID);
101 
102  // Create a vector of faces in the image
103  int numOfFaces;
104  ifs>>numOfFaces;
105  std::vector<cv::Rect> faces;
106  for(int i=0;i<numOfFaces;i++){
107  // Convert elipse to rectangle
108  double elipseX,elipseY,elipseTheta,elipseXRadious,elipseYRadious,confidence;
109  ifs>>elipseYRadious;
110  ifs>>elipseXRadious;
111  ifs>>elipseTheta;
112  ifs>>elipseX;
113  ifs>>elipseY;
114  ifs>>confidence;
115 
116  double rectX,rectY,rectWidth,rectHeight;
117  rectX=elipseX-elipseXRadious;
118  rectY=elipseY-elipseYRadious;
119  rectWidth=2*elipseXRadious;
120  rectHeight=2*elipseYRadious;
121 
122  faces.push_back(cv::Rect(static_cast<int>(rectX),
123  static_cast<int>(rectY),
124  static_cast<int>(rectWidth),
125  static_cast<int>(rectHeight))
126  );
127  }
128  std::pair<std::string,std::vector<cv::Rect>> p(sampID,faces);
129  m_objects.insert(p);
130  }
131  }
132  }
133 }
134 }
