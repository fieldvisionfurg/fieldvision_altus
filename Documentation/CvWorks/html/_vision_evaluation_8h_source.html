422  DetectorEvalResult computeResult() {
423 
424  DetectorEvalResult result;
425 
426  for(TestPoint& p : data){ //para cada testPoint
427 
428  // CALCULA A SIMILARIDADE DE CADA DETECÇÃO PARA CADA GROUNDTRUTH
429  int countD=0;
430  int countGT=0;
431  const unsigned sizeD = p.first.size();
432  const unsigned sizeGT = p.second.size();
433  result.detectionCount+=sizeD;
434  result.groundTruthCount+=sizeGT;
435  //W é uma matriz de similaridade onde as linhas são relacionadas as detecções e as colunas aos groud truths
436  double* W = new double[sizeD*sizeGT];
437  for(TObj1& d : p.first){ //para cada detecção
438  countGT=0;
439  for(TObj2& gt : p.second){ //para cada groundTruth
440  double similarity = simFcn(d,gt);
441  W[countD*sizeGT+countGT]=similarity;
442  countGT++;
443  }
444  countD++;
445  }
446 
447  // ATRELA UMA DETECÇÃO A UM GROUNDTRUTH MAXIMIZANDO A SOMA DE SIMILARIDADES (CASAMENTOS ÓTIMOS)
448  /* Por exemplo, A=[2 -1 0] significa que a detecção 0 foi casada ao groundTruth 2, a detecção 1 não teve casamento
449  e a detecção 3 foi casada ao groundTruth 0*/
450  std::vector<int> A = computeAssignmentOptimal(W,sizeD,sizeGT);
451 
452 
453  // CALCULA A QUANTIDADE DE FALSOS POSITIVOS ABSOLUTOS E FALSOS NEGATIVOS ABSOLUTOS
454  unsigned int afp=std::count(A.begin(),A.end(),-1);
455  result.absFpCount+=afp;
456 
457  unsigned int afn=0;
458  for(unsigned int i=0;i<sizeGT;i++){
459  if(std::find(A.begin(),A.end(),i) == A.end())
460  afn++;
461  }
462  result.absFnCount+=afn;
463 
464  // ADICIONA SIMILARIDADES AO VETOR similarityScores
465  for(unsigned int detIndex=0;detIndex<sizeD;detIndex++){
466  int gtIndex=A.at(detIndex);
467  if(gtIndex>=0) {//se detecção foi atrelada
468  result.similarityScores.push_back(W[detIndex*sizeGT+gtIndex]);
469  result.meanSimilarity += W[detIndex*sizeGT+gtIndex];
470  }
471  }
472 
473  delete[] W;
474  }
475 
476  // CALCULA ESTATÍSTICAS
477  result.falseNegCount = result.absFnCount;
478  result.falsePosCount = result.absFpCount;
479  for(const double& s : result.similarityScores){
480  if(s<threshold){
481  result.falseNegCount++;
482  result.falsePosCount++;
483  }
484  else{
485  result.truePosCount++;
486  }
487  }
488 
489 
493  result.precision = (double) result.truePosCount / (double)result.detectionCount;
497  result.recall = ((double) result.groundTruthCount - (double)result.falseNegCount) / (double)result.groundTruthCount;
499  result.f1score = (double) 2*(double) result.truePosCount/ ((2*(double) result.truePosCount) + (double) result.falsePosCount + (double) result.falseNegCount);
501  result.fdr = (double) result.falsePosCount / ((double) result.falsePosCount + (double) result.truePosCount);
503  result.fnr = 1 - result.recall;
505  result.meanSimilarity = result.meanSimilarity/(double)result.detectionCount;
506  //return
507  return result;
508 
509  }

515  DetectorEvalResult computeResultManyToMany() {
516 
517  DetectorEvalResult result;
518 
519  for(TestPoint& p : data){ //For Each testPoint
520 
521  int countD=0;
522  int countGT=0;
523  const int sizeD = p.first.size();
524  const int sizeGT = p.second.size();
525  result.detectionCount+=sizeD;
526  result.groundTruthCount+=sizeGT;
527 
528  //W é uma matriz de similaridade onde as linhas são relacionadas as detecções e as colunas aos groud truths
529  double* W = new double[sizeD*sizeGT];
530  for(TObj1& d : p.first){ //for each detection
531  countGT=0;
532  for(TObj2& gt : p.second){ //for each item in the ground truth
533  double similarity = simFcn(d,gt);
534  W[countD*sizeGT+countGT]=similarity;
535  countGT++;
536  }
537  countD++;
538  }
539 
540  // COMPUTE THE ABSOLUTE FALSE POSITIVE AND ABSOLUTE FALSE NEGATIVE (NO SIMILARITY)
541  // Using a 1-to-N approach we check if each detection has an associated ground truth
542  for(int i = 0; i< sizeD; i++){
543  bool is_fp = true;
544  for(int j = 0; j < sizeGT; j++){
545  // has an associated ground truth annotation
546  if(W[i*sizeGT+j] > 0.0f){
547  result.meanSimilarity += W[i*sizeGT+j];
548  is_fp = false;
549  break;
550  }
551  }
552 
553  if(is_fp) // absolute false positive found
554  result.absFpCount ++;
555  }
556 
557  // Using a 1-to-N approach we check if each ground truth annotation has at least one associated detection
558  for(int j = 0; j < sizeGT; j++){
559  bool is_fn = true;
560  for(int i = 0; i< sizeD; i++){
561  // ground truth has an associated detection
562  if(W[i*sizeGT+j] > 0.0f){
563  is_fn = false;
564  break;
565  }
566  }
567 
568  if(is_fn) // absolute false negative found
569  result.absFnCount ++;
570  }
571 
572 
573  // COMPUTE THE TRUE POSITIVE, FALSE POSITIVE AND FALSE NEGATIVE (similarity < threshold)
574  // For each detection
575  for(int i = 0; i< sizeD; i++){
576  bool is_fp = true;
577  for(int j = 0; j < sizeGT; j++){
578  //if the intersection is greater or equal than the defined threshold
579  if(W[i*sizeGT+j] >= threshold){
580  is_fp = false;
581  break;
582  }
583  }
584 
585  if(is_fp)
586  result.falsePosCount ++;
587  else // is detected and has the necessary similarity
588  result.truePosCount ++;
589  }
590 
592  for(int j = 0; j < sizeGT; j++){
593  bool is_fn = true;
594  for(int i = 0; i< sizeD; i++){
595  //if the intersection is greater or equal than the defined threshold
596  if(W[i*sizeGT+j] >= threshold){
597  is_fn = false;
598  break;
599  }
600  }
601 
602  if(is_fn){
603  result.falseNegCount ++;
604  }
605  }
606 
607  } // end of the test point loop
608 
612  result.precision = (double) result.truePosCount / (double)result.detectionCount;
616  result.recall = ((double) result.groundTruthCount - (double)result.falseNegCount) / (double)result.groundTruthCount;
618  result.f1score = (double) 2*(double) result.truePosCount/ ((2*(double) result.truePosCount) + (double) result.falsePosCount + (double) result.falseNegCount);
620  result.fdr = (double) result.falsePosCount / ((double) result.falsePosCount + (double) result.truePosCount);
622  result.fnr = 1 - result.recall;
624  result.meanSimilarity = result.meanSimilarity/(double)result.detectionCount;
625  //return
626  return result;
627  }

689  static std::vector<int> computeAssignmentOptimal(double *W,int sizeD,int sizeGT){
690  std::vector<int> A(sizeD,-1); //"assignment" ou atrelação para cada detecção
691 
692  // O algoritmo não funciona quando o número de linhas é maior que de colunas, então preenche com zeros (e o formato de matriz)
693  double** m = array_to_matrix(W,sizeD,sizeGT);
694  int rowsM = sizeD;
695  int colsM = sizeGT;
696  if(sizeD>sizeGT)
697  colsM=rowsM;
698 
699  hungarian_t p;
700 
701  /* initialize the gungarian_problem using the cost matrix*/
702  hungarian_init(&p, m, rowsM,colsM, HUNGARIAN_MAX);
703 
704  hungarian_solve(&p);
705 
706  //hungarian_print_assignment(&p);
707 
708  for(int i=0;i<sizeD;i++){
709  int assignment = p.a[i];
710  if(assignment<=(sizeGT-1)){ // if a detection was not assigned to an added column
711  if(W[i*sizeGT+assignment]>0) // and the weight is not zero
712  A.at(i)=assignment;
713  }
714  }
715 
716  //TODO: delete m ?
717  hungarian_fini(&p);
718  return A;
719  }

724  std::vector<size_t> sort_indexes(const std::vector<double> &v) {
725 
726  // initialize original index locations
727  std::vector<size_t> idx(v.size());
728  for (size_t i = 0; i != idx.size(); ++i) idx[i] = i;
729 
730  // sort indexes based on comparing values in v
731  std::sort(idx.begin(), idx.end(),
732  [&v](size_t i1, size_t i2) {return v[i1] < v[i2];});
733 
734  return idx;
735  }

746  std::vector<int> computeAssignmentNonOptimal(double *W,int sizeD,int sizeGT){
747 
748  std::vector<int> A(sizeD,-1); //"assignment" ou atrelação para cada detecção
749 
750  // Para cada detecção, faz um sort das similaridades e acha o indice do groundTruth com maior valor que ainda não foi atrelado a outra detecção.
751  for(int i=0;i<sizeD;i++){
752  std::vector<double> similarities(&W[i*sizeGT],&W[i*sizeGT+sizeGT]);
753  std::vector<size_t> sortIndex = sort_indexes(similarities);
754 
755  for (std::vector<size_t>::reverse_iterator indexIt = sortIndex.rbegin(); indexIt!= sortIndex.rend(); ++indexIt){
756  //verifica se o groundTruth com indice *indexIt ainda não foi atrelado a uma detecção
757  if(std::find(A.begin(),A.end(),*indexIt) == A.end()){
758  //se a similaridade for zero, atrela a -1 significando que não houve matching com nenhum groundTruth
759  int index = int(*indexIt);
760  if(similarities.at(index) != 0.0)
761  A.at(i)=index;
762  break;
763  }
764  }
765  }
766  return A;
767  }

898  BinaryClassifierEvalResult computeResult() {
899  // Implement the binary classifier evaluation
900  BinaryClassifierEvalResult result;
901 
902  for(TestPoint& p : data){ //For Each testPoint
903  bool p_gt = p.second;
904  bool p_co = p.first;
905 
906  // Groud-truth statistics
907  result.detectionCount ++;
908  if(p_gt)
909  result.posCount ++;
910  else
911  result.negCount ++;
912 
913  // BinaryClassifier statistics
914  if(p_gt && p_co)
915  result.truePosCount ++;
916  else if (!p_gt && p_co)
917  result.falsePosCount ++;
918  else if (p_gt && !p_co)
919  result.falseNegCount ++;
920  else if (!p_gt && !p_co)
921  result.trueNegCount ++;
922 
923  } // end of the test point loop
924 
928  result.accuracy = (double)(result.truePosCount+result.trueNegCount)/(double)result.detectionCount;
929 
932  result.recall = (double)result.truePosCount/(double)result.posCount;
933
